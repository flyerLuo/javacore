    public static void main(String[] args) throws IOException {

        read(FILEPATH);
    }

    private static void write(String filepath) throws IOException {
        // 1.Сй┐уће File у▒╗у╗Љт«џСИђСИфТќЄС╗Х

            dos.writeChar('\n');

        dos.close();

    private static void read(String filepath) throws IOException {
        // 1.Сй┐уће File у▒╗у╗Љт«џСИђСИфТќЄС╗Х

        int len = 0; // С┐ЮтГўУ»╗тЈќТЋ░ТЇ«уџёСИфТЋ░

        try {
            while (true) {
                temp = new char[200]; // т╝ђУЙЪуЕ║жЌ┤
                len = 0;
                while ((c = dis.readChar()) != '\t') { // ТјЦТћХтєЁт«╣

        // 3.У┐ЏУАїУ»╗ТѕќтєЎТЊЇСйю
        int temp = 0; // ТјЦТћХТ»ЈСИђСИфтєЁт«╣
        int len = 0; // У»╗тЈќтєЁт«╣
        char[] c = new char[1024];
        while ((temp = input.read()) != -1) {
            // тдѓТъюСИЇТў»-1т░▒УАеуц║У┐ўТюЅтєЁт«╣№╝їтЈ»С╗Цу╗Ду╗ГУ»╗тЈќ
            c[len] = (char) temp;
            len++;
        }
        System.out.println("ТќЄС╗ХтГЌугдТЋ░СИ║№╝џ" + len);

        // 4.тЁ│жЌГТхЂ
        input.close();

        return c;
    }

}

        # тГЌУіѓТхЂУйгТЇбтГЌугдТхЂ

        ТѕЉС╗гтЈ»С╗ЦтюеуеІт║ЈСИГжђџУ┐Є InputStream тњї Reader С╗јТЋ░ТЇ«Т║љСИГУ»╗тЈќТЋ░ТЇ«№╝їуёХтљјС╣ЪтЈ»С╗ЦтюеуеІт║ЈСИГт░єТЋ░ТЇ«жђџУ┐Є OutputStream тњї Writer УЙЊтЄ║тѕ░уЏ«ТаЄтфњС╗ІСИГ

        Сй┐уће InputStreamReader тЈ»С╗Цт░єУЙЊтЁЦтГЌУіѓТхЂУйгтїќСИ║УЙЊтЁЦтГЌугдТхЂ№╝ЏСй┐ућеOutputStreamWriterтЈ»С╗Цт░єУЙЊтЄ║тГЌУіѓТхЂУйгтїќСИ║УЙЊтЄ║тГЌугдТхЂсђѓ

        OutputStreamWriter уц║СЙІ№╝џ

        public class OutputStreamWriterDemo {

    public static void main(String[] args) throws IOException {
        File f = new File("temp.log");
        Writer out = new OutputStreamWriter(new FileOutputStream(f));
        out.write("hello world!!");
        out.close();
    }

}

        InputStreamReader уц║СЙІ№╝џ

        public class InputStreamReaderDemo {

    public static void main(String[] args) throws IOException {
        File f = new File("temp.log");
        Reader reader = new InputStreamReader(new FileInputStream(f));
        char[] c = new char[1024];
        int len = reader.read(c);
        reader.close();
        System.out.println(new String(c, 0, len));
    }

}

        # тГЌУіѓТхЂ vs. тГЌугдТхЂ

        уЏИтљїуѓ╣№╝џ

        тГЌУіѓТхЂтњїтГЌугдТхЂжЃйТюЅ read()сђЂwrite()сђЂclose()сђЂflush() У┐ЎТаиуџёТќ╣Т│Ћ№╝їУ┐Ўтє│т«џС║єт«ЃС╗гуџёТЊЇСйюТќ╣т╝ЈУ┐ЉС╝╝сђѓ

        СИЇтљїуѓ╣№╝џ

        • ТЋ░ТЇ«у▒╗тъІ
          • тГЌУіѓТхЂСИ╗УдЂТЊЇСйютГЌУіѓТЋ░ТЇ«ТѕќС║їУ┐ЏтѕХт»╣У▒АсђѓСИ╗УдЂТаИт┐Ѓу▒╗Тў» InputStream у▒╗тњї OutputStream у▒╗сђѓ
          • тГЌугдТхЂСИ╗УдЂТЊЇСйютГЌугд№╝їСИђСИфтГЌугдуГЅС║јСИцСИфтГЌУіѓсђѓСИ╗УдЂТаИт┐Ѓу▒╗Тў» Reader у▒╗тњї Writer у▒╗сђѓ
        • у╝Њтє▓тї║
          • тГЌУіѓТхЂтюеТЊЇСйюТЌХТюгУ║ФСИЇС╝џућетѕ░у╝Њтє▓тї║№╝ѕтєЁтГў№╝Ѕ№╝їТў»ТќЄС╗ХТюгУ║ФуЏ┤ТјЦТЊЇСйюуџёсђѓ
          • тГЌугдТхЂтюеТЊЇСйюТЌХТў»Сй┐ућеС║єу╝Њтє▓тї║№╝їжђџУ┐Єу╝Њтє▓тї║тєЇТЊЇСйюТќЄС╗Хсђѓ

        жђЅТІЕ№╝џ

        ТЅђТюЅуџёТќЄС╗ХтюеуАгуЏўТѕќС╝аУЙЊТЌХжЃйТў»С╗ЦтГЌУіѓТќ╣т╝ЈС┐ЮтГўуџё№╝їСЙІтдѓтЏЙуЅЄ№╝їтй▒жЪ│ТќЄС╗ХуГЅжЃйТў»ТїЅтГЌУіѓТќ╣т╝ЈтГўтѓеуџёсђѓтГЌугдТхЂТЌаТ│ЋУ»╗тєЎУ┐ЎС║ЏТќЄС╗Хсђѓ

        ТЅђС╗Ц№╝їжЎцС║єу║»ТќЄТюгТЋ░ТЇ«ТќЄС╗ХСй┐ућетГЌугдТхЂС╗Цтцќ№╝їтЁХС╗ќТќЄС╗Ху▒╗тъІжЃйт║ћУ»ЦСй┐ућетГЌУіѓТхЂТќ╣т╝Јсђѓ

        # С║їсђЂС╝ау╗Ъ BIO

        тюе Linux СИГ№╝їтйЊт║ћућеУ┐ЏуеІУ░Ѓуће recvfrom Тќ╣Т│ЋУ░ЃућеТЋ░ТЇ«уџёТЌХтђЎ№╝їтдѓТъютєЁТаИТ▓АТюЅТііТЋ░ТЇ«тЄєтцЄтЦйСИЇС╝џуФІтѕ╗У┐ћтЏъ№╝їУђїТў»С╝џу╗ЈтјєуГЅтЙЁТЋ░ТЇ«тЄєтцЄт░▒у╗ф№╝їТЋ░ТЇ«С╗јтєЁТаИтцЇтѕХтѕ░ућеТѕиуЕ║жЌ┤С╣ІтљјтєЇУ┐ћтЏъ№╝їУ┐ЎТюЪжЌ┤т║ћућеУ┐ЏуеІСИђуЏ┤жў╗тАъуЏ┤тѕ░У┐ћтЏъ№╝їТЅђС╗ЦУбФуД░СИ║жў╗тАъ IO ТеАтъІсђѓ

        жў╗тАъ IO ТеАтъІ СИЇжђѓућежФўт╣ХтЈЉтю║ТЎ»сђѓтЏаСИ║СИђСИфУ»иТ▒ѓ IO С╝џжў╗тАъУ┐ЏуеІ№╝їТЅђС╗Ц№╝їтЙЌСИ║Т»ЈУ»иТ▒ѓтѕєжЁЇСИђСИфтцёуљєУ┐ЏуеІ№╝ѕу║┐уеІ№╝ЅС╗ЦтЈіТЌХтЊЇт║ћ№╝їу│╗у╗Ът╝ђжћђтцДсђѓ

        жЄЄуће BIO жђџС┐АТеАтъІ уџёТюЇтіАуФ»№╝їжђџтИИућ▒СИђСИфуІгуФІуџё Acceptor у║┐уеІУ┤ЪУ┤БуЏЉтљгт«бТѕиуФ»уџёУ┐ъТјЦсђѓТѕЉС╗гСИђУѕгжђџУ┐Єтюеwhile(true) тЙфуј»СИГТюЇтіАуФ»С╝џУ░Ѓуће accept() Тќ╣Т│ЋуГЅтЙЁТјЦТћХт«бТѕиуФ»уџёУ┐ъТјЦуџёТќ╣т╝ЈуЏЉтљгУ»иТ▒ѓ№╝їУ»иТ▒ѓСИђТЌдТјЦТћХтѕ░СИђСИфУ┐ъТјЦУ»иТ▒ѓ№╝їт░▒тЈ»С╗Цт╗║уФІжђџС┐АтЦЌТјЦтГЌтюеУ┐ЎСИфжђџС┐АтЦЌТјЦтГЌСИіУ┐ЏУАїУ»╗тєЎТЊЇСйю№╝їТГцТЌХСИЇУЃйтєЇТјЦТћХтЁХС╗ќт«бТѕиуФ»У┐ъТјЦУ»иТ▒ѓ№╝їтЈфУЃйуГЅтЙЁтљїтйЊтЅЇУ┐ъТјЦуџёт«бТѕиуФ»уџёТЊЇСйюТЅДУАїт«їТѕљ№╝ї СИЇУ┐ЄтЈ»С╗ЦжђџУ┐Єтцџу║┐уеІТЮЦТћ»ТїЂтцџСИфт«бТѕиуФ»уџёУ┐ъТјЦ№╝їтдѓСИітЏЙТЅђуц║сђѓ

        тдѓТъюУдЂУ«Е BIO жђџС┐АТеАтъІ УЃйтцЪтљїТЌХтцёуљєтцџСИфт«бТѕиуФ»У»иТ▒ѓ№╝їт░▒т┐ЁжА╗Сй┐ућетцџу║┐уеІ№╝ѕСИ╗УдЂтјЪтЏаТў»socket.accept()сђЂsocket.read()сђЂsocket.write() ТХЅтЈіуџёСИЅСИфСИ╗УдЂтЄйТЋ░жЃйТў»тљїТГЦжў╗тАъуџё№╝Ѕ№╝їС╣Ът░▒Тў»У»┤т«ЃтюеТјЦТћХтѕ░т«бТѕиуФ»У┐ъТјЦУ»иТ▒ѓС╣ІтљјСИ║Т»ЈСИфт«бТѕиуФ»тѕЏт╗║СИђСИфТќ░уџёу║┐уеІУ┐ЏУАїжЊЙУи»тцёуљє№╝їтцёуљєт«їТѕљС╣Ітљј№╝їжђџУ┐ЄУЙЊтЄ║ТхЂУ┐ћтЏът║ћуГћу╗Ўт«бТѕиуФ»№╝їу║┐уеІжћђТ»ЂсђѓУ┐Ўт░▒Тў»тЁИтъІуџё СИђУ»иТ▒ѓСИђт║ћуГћжђџС┐АТеАтъІ сђѓТѕЉС╗гтЈ»С╗ЦУ«ЙТЃ│СИђСИІтдѓТъюУ┐ЎСИфУ┐ъТјЦСИЇтЂџС╗╗СйЋС║ІТЃЁуџёУ»Ют░▒С╝џжђаТѕљСИЇт┐ЁУдЂуџёу║┐уеІт╝ђжћђ№╝їСИЇУ┐ЄтЈ»С╗ЦжђџУ┐Є у║┐уеІТ▒аТю║тѕХ Тћ╣тќё№╝їу║┐уеІТ▒аУ┐ўтЈ»С╗ЦУ«Еу║┐уеІуџётѕЏт╗║тњїтЏъТћХТѕљТюгуЏИт»╣УЙЃСйјсђѓСй┐ућеFixedThreadPool тЈ»С╗ЦТюЅТЋѕуџёТјДтѕХС║єу║┐уеІуџёТюђтцДТЋ░жЄЈ№╝їС┐ЮУ»ЂС║єу│╗у╗ЪТюЅжЎљуџёУхёТ║љуџёТјДтѕХ№╝їт«ъуј░С║є N(т«бТѕиуФ»У»иТ▒ѓТЋ░жЄЈ):M(тцёуљєт«бТѕиуФ»У»иТ▒ѓуџёу║┐уеІТЋ░жЄЈ)уџёС╝фт╝ѓТГЦ I/O ТеАтъІ№╝ѕN тЈ»С╗ЦУ┐юУ┐ютцДС║ј M№╝Ѕ№╝їСИІжЮбСИђУіѓ"С╝фт╝ѓТГЦ BIO"СИГС╝џУ»ду╗єС╗Іу╗Їтѕ░сђѓ

        ТѕЉС╗гтєЇУ«ЙТЃ│СИђСИІтйЊт«бТѕиуФ»т╣ХтЈЉУ«┐жЌ«жЄЈтбътіатљјУ┐ЎуДЇТеАтъІС╝џтЄ║уј░С╗ђС╣ѕжЌ«жбў№╝Ъ

        тюе Java УЎџТІЪТю║СИГ№╝їу║┐уеІТў»т«ЮУ┤хуџёУхёТ║љ№╝їу║┐уеІуџётѕЏт╗║тњїжћђТ»ЂТѕљТюгтЙѕжФў№╝їжЎцТГцС╣Ітцќ№╝їу║┐уеІуџётѕЄТЇбТѕљТюгС╣ЪТў»тЙѕжФўуџёсђѓт░цтЁХтюе Linux У┐ЎТаиуџёТЊЇСйюу│╗у╗ЪСИГ№╝їу║┐уеІТюгУ┤еСИіт░▒Тў»СИђСИфУ┐ЏуеІ№╝їтѕЏт╗║тњїжћђТ»Ђу║┐уеІжЃйТў»жЄЇжЄЈу║Дуџёу│╗у╗ЪтЄйТЋ░сђѓтдѓТъют╣ХтЈЉУ«┐жЌ«жЄЈтбътіаС╝џт»╝УЄ┤у║┐уеІТЋ░ТђЦтЅДУєеУЃђтЈ»УЃйС╝џт»╝УЄ┤у║┐уеІтаєТаѕТ║бтЄ║сђЂтѕЏт╗║Тќ░у║┐уеІтц▒У┤ЦуГЅжЌ«жбў№╝їТюђу╗ѕт»╝УЄ┤У┐ЏуеІт«ЋТю║ТѕќУђЁтЃхТГ╗№╝їСИЇУЃйт»╣тцќТЈљСЙЏТюЇтіАсђѓ

        # СИЅсђЂС╝фт╝ѓТГЦ BIO

        СИ║С║єУДБтє│тљїТГЦжў╗тАъ I/O жЮбСИ┤уџёСИђСИфжЊЙУи»жюђУдЂСИђСИфу║┐уеІтцёуљєуџёжЌ«жбў№╝їтљјТЮЦТюЅС║║т»╣т«Ѓуџёу║┐уеІТеАтъІУ┐ЏУАїС║єС╝ўтїќСИђСИђСИђтљјуФ»жђџУ┐ЄСИђСИфу║┐уеІТ▒аТЮЦтцёуљєтцџСИфт«бТѕиуФ»уџёУ»иТ▒ѓТјЦтЁЦ№╝їтйбТѕљт«бТѕиуФ»СИфТЋ░ M№╝џу║┐уеІТ▒аТюђтцДу║┐уеІТЋ░ N уџёТ»ћСЙІтЁ│у│╗№╝їтЁХСИГ M тЈ»С╗ЦУ┐юУ┐ютцДС║ј NсђѓжђџУ┐Єу║┐уеІТ▒атЈ»С╗ЦуЂхТ┤╗тю░У░ЃжЁЇу║┐уеІУхёТ║љ№╝їУ«Йуй«у║┐уеІуџёТюђтцДтђ╝№╝їжў▓ТГбућ▒С║јТхижЄЈт╣ХтЈЉТјЦтЁЦт»╝УЄ┤у║┐уеІУђЌт░йсђѓ

        жЄЄућеу║┐уеІТ▒атњїС╗╗тіАжўЪтѕЌтЈ»С╗Цт«ъуј░СИђуДЇтЈФтЂџС╝фт╝ѓТГЦуџё I/O жђџС┐АТАєТъХ№╝їт«ЃуџёТеАтъІтЏЙтдѓСИітЏЙТЅђуц║сђѓтйЊТюЅТќ░уџёт«бТѕиуФ»ТјЦтЁЦТЌХ№╝їт░єт«бТѕиуФ»уџё Socket т░ЂУБЁТѕљСИђСИф Task№╝ѕУ»ЦС╗╗тіАт«ъуј░ java.lang.Runnable ТјЦтЈБ№╝ЅТіЋжђњтѕ░тљјуФ»уџёу║┐уеІТ▒аСИГУ┐ЏУАїтцёуљє№╝їJDK уџёу║┐уеІТ▒ау╗┤ТіцСИђСИфТХѕТЂ»жўЪтѕЌтњї N СИфТ┤╗УиЃу║┐уеІ№╝їт»╣ТХѕТЂ»жўЪтѕЌСИГуџёС╗╗тіАУ┐ЏУАїтцёуљєсђѓућ▒С║ју║┐уеІТ▒атЈ»С╗ЦУ«Йуй«ТХѕТЂ»жўЪтѕЌуџётцДт░ЈтњїТюђтцДу║┐уеІТЋ░№╝їтЏаТГц№╝їт«ЃуџёУхёТ║љтЇаућеТў»тЈ»ТјДуџё№╝їТЌаУ«║тцџт░ЉСИфт«бТѕиуФ»т╣ХтЈЉУ«┐жЌ«№╝їжЃйСИЇС╝џт»╝УЄ┤УхёТ║љуџёУђЌт░йтњїт«ЋТю║сђѓ

        С╝фт╝ѓТГЦ I/O жђџС┐АТАєТъХжЄЄућеС║єу║┐уеІТ▒ат«ъуј░№╝їтЏаТГцжЂ┐тЁЇС║єСИ║Т»ЈСИфУ»иТ▒ѓжЃйтѕЏт╗║СИђСИфуІгуФІу║┐уеІжђаТѕљуџёу║┐уеІУхёТ║љУђЌт░йжЌ«жбўсђѓСИЇУ┐ЄтЏаСИ║т«Ѓуџёт║Ћт▒ѓС╗ЇуёХТў»тљїТГЦжў╗тАъуџё BIO ТеАтъІ№╝їтЏаТГцТЌаТ│ЋС╗јТа╣ТюгСИіУДБтє│жЌ«жбўсђѓ
